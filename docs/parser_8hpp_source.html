37  int parse_value(arg& argu, char* argv[], int counter) {
38  // next arg isn't a flag or option
39  // intersperse values with a space separator
40  if(argv[counter] != nullptr && argv[counter][0] != '-') {
41  // overwrite value
42  argu.value = argv[counter];
43  ++counter;
44  }
45 
46  while(argv[counter] != nullptr && argv[counter][0] != '-') {
47  // append the value to the space separated list
48  argu.value.push_back(' ');
49  argu.value.append(argv[counter++]);
50  }
51 
52  return counter - 1;
53  }

55  void parse_short_option(const std::string& option, int& i, char* argv[]) {
56  for(auto&& arg : args) {
57 
58  auto pos = option.find(arg.second.short_name);
59 
60  // Single flag or option with value
61  if(option.size() == 2) {
62  if(pos != std::string::npos) {
63  if(arg.second.is_value()) {
64  i = parse_value(arg.second, argv, i + 1);
65  break;
66  }
67 
68  // regular flag
69 
70  arg.second.active = true;
71  }
72  }
73 
74  // Grouped character or grouped option
75  else if(option.size() > 2) {
76 
77  // Grouped option
78  if(arg.second.is_value() && option[1] == arg.second.short_name) {
79  arg.second.value.append(std::begin(option) + 2, std::end(option));
80  arg.second.active = true;
81  break;
82  }
83 
84  // regular flag with grouped characters
85  arg.second.active = pos != std::string::npos;
86  }
87  else {
88  // Seems to be neither
89  break;
90  }
91  }
92  }

114  void parse(int argc, char* argv[]) {
115  std::string current;
116 
117  if(program_name.empty()) {
118  program_name = argv[0];
119  }
120 
121  for(int i = 1; i < argc; ++i) {
122  current = argv[i];
123 
124  // Stop parsing if it's --
125  if(current == "--")
126  break;
127 
128  // current arg has equal sign
129  auto pos = current.find('=');
130  if(pos != std::string::npos) {
131  auto it = args.find(current.substr(0, pos));
132 
133  // make sure the argument is valid
134  if(it != args.end()) {
135  // Override value
136  it->second.value = current.substr(pos + 1);
137  }
138 
139  continue;
140  }
141 
142  auto it = args.find(current);
143 
144  if(it != args.end()) {
145 
146  // value options
147  if(it->second.is_value()) {
148  i = parse_value(it->second, argv, i + 1);
149  }
150 
151  // regular flag
152 
153  it->second.active = true;
154  continue;
155  }
156 
157  // short options found
158  if(current.front() == '-') {
159  parse_short_option(current, i, argv);
160  }
161  else {
162  // otherwise, it is an unrecognised argument
163  throw detail::unrecognised_arg(program_name, current);
164  }
165  }
166  }

186  template<typename T = std::string>
187  T get(const std::string& str) const {
188  const auto& argument = "--" + str;
189  auto it = args.find(argument);
190  if(it != args.end()) {
191  if(it->second.is_value())
192  return detail::lexical_cast<T>(it->second.value);
193  }
194 
195  throw detail::invalid_arg_get(argument);
196  }

198  template<typename T>
199  T get_list(const std::string& str) const {
200  const auto& argument = "--" + str;
201  auto it = args.find(argument);
202  if(it != args.end()) {
203  if(it->second.is_value()) {
204  T result;
205  std::istringstream ss(it->second.value);
206  typename T::value_type x;
207  while(ss >> x) {
208  detail::insert_impl(result, x);
209  }
210 
211  return result;
212  }
213  }
214 
215  throw detail::invalid_arg_get(argument);
216  }

218  void check_required_args() const {
219  // loop through every argument to see if the required
220  // arguments haven't been provided.
221  for(auto&& arg : args) {
222  if(arg.second.req && !arg.second.active) {
223  throw detail::missing_required_arg(program_name, "--" + arg.second.name);
