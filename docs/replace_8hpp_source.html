31 template<typename String>
32 inline size_t nth_finder(const String& str, const String& find, size_t nth) {
33  size_t start = 0;
34  auto pos = str.find(find);
35  while(start < nth && pos != String::npos) {
36  pos = str.find(find, pos + find.length());
37  ++start;
38  }
39  return pos;
40 }
