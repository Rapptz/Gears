39 class any {
40 private:
41  template<typename T>
42  using Decayed = typename std::decay<T>::type;
43 
44  struct base {
45  virtual ~base() {}
46  virtual std::unique_ptr<base> clone() const = 0;
47  };
48 
49  std::unique_ptr<base> obj;
50 
51  template<typename T>
52  struct object : base {
53  static_assert(std::is_copy_constructible<Decayed<T>>(), "Concept violation: T is not CopyConstructible");
54  template<typename U>
55  object(U&& v): value(std::forward<U>(v)) {}
56  T value;
57 
58  std::unique_ptr<base> clone() const {
59  return std::unique_ptr<base>(new object<T>(value));
60  }
61  };
62 
63  std::unique_ptr<base> clone() const {
64  return obj ? obj->clone() : nullptr;
65  }
66 public:
67  any() noexcept = default;
68  template<typename T>
69  any(T&& actual) noexcept: obj(new(std::nothrow) object<Decayed<T>>(std::forward<T>(actual))) {}
70  any(any&) noexcept = default;
71  any(const any&) noexcept = default;
72  any(any&&) noexcept = default;
73 
74  explicit operator bool() const {
75  return static_cast<bool>(obj);
76  }
77 
78  template<typename T>
79  bool is() const {
80  auto ptr = dynamic_cast<object<Decayed<T>>*>(obj.get());
81  return ptr;
82  }
83 
84  template<typename T>
85  Decayed<T>& as() {
86  auto ptr = dynamic_cast<object<Decayed<T>>*>(obj.get());
87  if(!ptr)
88  throw any_detail::bad_any_cast();
89  return ptr->value;
90  }
91 
92  template<typename T>
93  const Decayed<T>& as() const {
94  auto ptr = dynamic_cast<object<Decayed<T>>*>(obj.get());
95  if(!ptr)
96  throw any_detail::bad_any_cast();
97  return ptr->value;
98  }
99 
100  any& operator=(const any& other) {
101  if(obj == other.obj)
102  return *this;
103 
104  obj = other.clone();
105  return *this;
106  }
107 
108  any& operator=(any&& other) {
109  if(obj == other.obj)
110  return *this;
111 
112  obj.swap(other.obj);
113  return *this;
114  }
115 };
