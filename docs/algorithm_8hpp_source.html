40 inline T fibonacci(T number) {
41  static_assert(std::is_integral<T>::value, "Type passed must be an integral type");
42  if(number < 2) {
43  return number;
44  }
45 
46  T a(1);
47  T b(0);
48  T c(0);
49 
50  while(--number) {
51  c = b;
52  b = a;
53  a = b + c;
54  }
55 
56  return a;
57 }

104 template<typename T>
105 inline T mod_pow(T base, T exponent, const T& modulus) {
106  static_assert(std::is_integral<T>::value, "Type passed must be an integral type");
107  T result(1);
108 
109  while(exponent) {
110  if((exponent % 2) == 1) {
111  result = (base * result) % modulus;
112  }
113  exponent /= 2;
114  base = (base * base) % modulus;
115  }
116 
117  return result;
118 }

131 template<typename T>
132 inline T sum_of_divisors(const T& number) noexcept {
133  static_assert(std::is_integral<T>::value, "Type passed must be an integral type");
134  T result(0);
135  for(T i(1); i * i <= number; ++i) {
136  result += (number % i) ? 0 : ((i * i == number) ? i : i + number / i);
137  }
138  return result;
139 }
