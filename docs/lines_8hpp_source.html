31 template<typename CharT = char, typename Traits = std::char_traits<CharT>>
32 struct line_iterator : std::iterator<std::input_iterator_tag, std::basic_string<CharT, Traits>> {
33 private:
34  std::basic_istream<CharT, Traits>* reader;
35  std::basic_string<CharT, Traits> value;
36  bool status;
37 public:
38  line_iterator() noexcept: reader(nullptr), status(false) {}
39  line_iterator(std::basic_istream<CharT, Traits>& out) noexcept: reader(&out) {
40  status = reader && *reader && std::getline(*reader, value);
41  }
42 
43  auto operator++() noexcept -> decltype(*this) {
44  status = reader && *reader && std::getline(*reader, value);
45  return *this;
46  }
47 
48  line_iterator operator++(int) noexcept {
49  auto copy = *this;
50  ++(*this);
51  return copy;
52  }
53 
54  auto operator*() noexcept -> decltype(value) {
55  return value;
56  }
57 
58  auto operator->() noexcept -> decltype(&value) {
59  return &value;
60  }
61 
62  bool operator==(const line_iterator& other) const noexcept {
63  return (status == other.status) && (!status || reader == other.reader);
64  }
65 
66  bool operator!=(const line_iterator& other) const noexcept {
67  return not (*this == other);
68  }
69 };
